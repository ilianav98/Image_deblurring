import os
import cv2
import numpy as np

blurred_dir = 'noisy_images'
sharp_dir = 'original_images'


image_size = (64, 64)

def load_images_from_directory(directory, image_size):
    images = []
    for filename in os.listdir(directory):
        img_path = os.path.join(directory, filename)
        img = cv2.imread(img_path)  # Read the image
        img = cv2.resize(img, image_size)  # Resize to the required dimensions
        img = img.astype('float32') / 255.0  # Normalize to [0, 1]
        images.append(img)
    return np.array(images)

# Loading blurred and sharp images
blurred_images = load_images_from_directory(blurred_dir, image_size)
sharp_images = load_images_from_directory(sharp_dir, image_size)

from skimage.metrics import peak_signal_noise_ratio as psnr

# Calculating PSNR between sharp images and reconstructed images
def calculate_psnr(sharp_images, reconstructed_images):
    psnr_values = []
    for i in range(len(sharp_images)):
        psnr_value = psnr(sharp_images[i], reconstructed_images[i], data_range=1.0)
        psnr_values.append(psnr_value)
    
    avg_psnr = np.mean(psnr_values)
    return avg_psnr

# Calculating the average PSNR for the dataset
average_psnr = calculate_psnr(sharp_images, reconstructed_images)
print(f'Average PSNR: {average_psnr:.2f} dB')

import matplotlib.pyplot as plt

def display_image_comparison(blurred_images, sharp_images, reconstructed_images, num_images=5):
    plt.figure(figsize=(15, 5))
    
    for i in range(num_images):
        # Displaying blurred image
        ax = plt.subplot(3, num_images, i + 1)
        plt.imshow(blurred_images[i])
        plt.title("Blurred")
        plt.axis("off")
        
        # Displaying reconstructed image
        ax = plt.subplot(3, num_images, i + 1 + num_images)
        plt.imshow(reconstructed_images[i])
        plt.title("Reconstructed")
        plt.axis("off")
        
        # Displaying sharp image
        ax = plt.subplot(3, num_images, i + 1 + 2 * num_images)
        plt.imshow(sharp_images[i])
        plt.title("Sharp")
        plt.axis("off")
    
    plt.show()
